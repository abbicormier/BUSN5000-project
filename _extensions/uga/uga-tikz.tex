% Draw a legend
% Usage: \ugalegend{x}{y}{{label1/color1},{label2/color2},...}
\newcommand{\ugalegend}[3]{
    \begin{scope}[shift={(#1,#2)}]
        \foreach \label/\col [count=\i from 0] in {#3} {
            \fill[\col] (0,-\i*0.5) rectangle (0.4,-\i*0.5-0.3);
            \node[right, font=\footnotesize] at (0.5,-\i*0.5-0.15) {\label};
        }
    \end{scope}
}

% Draw a scale bar
% Usage: \ugascalebar{x}{y}{length}{label}
\newcommand{\ugascalebar}[4]{
    \draw[thick, dawgRed] (#1,#2) -- ++(#3,0);
    \draw[thick, dawgRed] (#1,#2-0.1) -- ++(0,0.2);
    \draw[thick, dawgRed] (#1+#3,#2-0.1) -- ++(0,0.2);
    \node[below, font=\footnotesize] at (#1+#3/2,#2-0.15) {#4};
}
